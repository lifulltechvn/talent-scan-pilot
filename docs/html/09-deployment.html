Local
Production (EC2)
# 1. Clone & config git clone <repo-url> cd talent-scan-pilot cp .env.example .env # Sửa .env: AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY # 2. Start docker compose up -d # 3. Database docker compose exec api alembic upgrade head # 4. Seed (optional) docker compose exec api python seed.py # → http://localhost (hr@test.com / test1234)
# 1. Launch EC2 (Ubuntu 22.04, t2.medium) ssh -i key.pem ubuntu@<ip> # 2. Install Docker curl -fsSL https://get.docker.com | sh sudo usermod -aG docker ubuntu # 3. Clone & config git clone <repo-url> cd talent-scan-pilot cp .env.example .env # Set production values: DATABASE_URL, SECRET_KEY, AWS_* # 4. Build & start ./build.sh docker compose up -d docker compose exec api alembic upgrade head # 5. SSL (optional — Certbot) sudo apt install certbot python3-certbot-nginx sudo certbot --nginx -d yourdomain.com
